   699тЖТ   699тЖТ// тФАтФАтФА Permission Checker тФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФА
   700тЖТ   700тЖТ
   701тЖТ   701тЖТexport class PermissionChecker {
   702тЖТ   702тЖТ  /**
   703тЖТ   703тЖТ   * Check if a set of roles grants a specific permission.
   704тЖТ   704тЖТ   */
   705тЖТ   705тЖТ  static hasPermission(roles: string[], permission: Permission): boolean {
   706тЖТ   706тЖТ    for (const role of roles) {
   707тЖТ   707тЖТ      const rolePermissions = DEFAULT_ROLES[role]
   708тЖТ   708тЖТ      if (rolePermissions?.includes(permission)) return true
   709тЖТ   709тЖТ    }
   710тЖТ   710тЖТ    return false
   711тЖТ   711тЖТ  }
   712тЖТ   712тЖТ
   713тЖТ   713тЖТ  /**
   714тЖТ   714тЖТ   * Check if user has ANY of the given permissions.
   715тЖТ   715тЖТ   */
   716тЖТ   716тЖТ  static hasAnyPermission(roles: string[], permissions: Permission[]): boolean {
   717тЖТ   717тЖТ    return permissions.some((p) => this.hasPermission(roles, p))
   718тЖТ   718тЖТ  }
   719тЖТ   719тЖТ
   720тЖТ   720тЖТ  /**
   721тЖТ   721тЖТ   * Check if user has ALL of the given permissions.
   722тЖТ   722тЖТ   */
   723тЖТ   723тЖТ  static hasAllPermissions(roles: string[], permissions: Permission[]): boolean {
   724тЖТ   724тЖТ    return permissions.every((p) => this.hasPermission(roles, p))
   725тЖТ   725тЖТ  }
   726тЖТ   726тЖТ
   727тЖТ   727тЖТ  /**
   728тЖТ   728тЖТ   * Resolve all permissions for a set of roles (union).
   729тЖТ   729тЖТ   */
   730тЖТ   730тЖТ  static resolvePermissions(roles: string[]): string[] {
   731тЖТ   731тЖТ    const result = new Set<string>()
   732тЖТ   732тЖТ    for (const role of roles) {
   733тЖТ   733тЖТ      const perms = DEFAULT_ROLES[role]
   734тЖТ   734тЖТ      if (perms) perms.forEach((p) => result.add(p))
   735тЖТ   735тЖТ    }
   736тЖТ   736тЖТ    return Array.from(result)
   737тЖТ   737тЖТ  }
   738тЖТ   738тЖТ
   739тЖТ   739тЖТ  /**
   740тЖТ   740тЖТ   * Check if two roles conflict (SoD violation).
   741тЖТ   741тЖТ   * Returns true if assigning both roles to the same user would violate SoD.
   742тЖТ   742тЖТ   */
   743тЖТ   743тЖТ  static isRoleConflict(roleA: string, roleB: string): boolean {
   744тЖТ   744тЖТ    const CONFLICTING_PAIRS: Array<[string, string]> = [
   745тЖТ   745тЖТ      ['finance_accountant', 'finance_manager'],     // SoD-06: JE creator cannot post
   746тЖТ   746тЖТ      ['procurement_officer', 'procurement_manager'], // SoD-01: PR creator cannot approve
   747тЖТ   747тЖТ      ['sales_officer', 'sales_manager'],             // SoD-11: SO creator cannot approve
